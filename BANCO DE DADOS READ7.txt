CREATE SCHEMA BD;
USE BD;

CREATE TABLE Usuarios(
nome VARCHAR(200),
senha CHAR(10),
email VARCHAR(100),
ID INT NOT NULL AUTO_INCREMENT PRIMARY KEY
);

CREATE TABLE ADM (
nome VARCHAR(200) NOT NULL,
senha CHAR(10) NOT NULL
);

CREATE TABLE Resumos (
ID_Resumo INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
ID_Usuário INT,
emailAutor VARCHAR(100),
Título VARCHAR(100),
TextoUsuario TEXT,
Classificação_resumo CHAR(20),
FOREIGN KEY (ID_Usuário) REFERENCES Usuarios(ID)
);
